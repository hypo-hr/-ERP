       if(!t) return;
       let v = t.value.replace(/[^0-9]/g, '');
       const bank = document.getElementById('formBank')?.value;
       
       const formats = {
           'KB국민은행': [6, 2, 6], '신한은행': [3, 3, 6], '우리은행': [4, 3, 6],
           '하나은행': [3, 6, 5], 'IBK기업은행': [3, 6, 2, 2], 'NH농협은행': [3, 4, 4, 2],

       };
       
       let fmt = formats[bank];
       if (fmt) {
           let res = ''; let idx = 0;
           for (let i=0; i<fmt.length; i++) {
               if (v.length > idx) {
                   if (i > 0) res += '-';
                   res += v.slice(idx, idx + fmt[i]);
                   idx += fmt[i];
               }
           }
           if (v.length > idx) res += v.slice(idx);
           t.value = res;
       } else {
           t.value = v;
       }
   };

       const f = emps.filter(e => ['정규직','계약직','일용직'].includes(e.contractType) && (e.name.includes(s.value) || e.empID.includes(s.value)));
       b.innerHTML = f.map(e => `<tr onclick="window.selectEmpForSalary('${e.empID}', this)"><td>${e.empID}</td><td>${e.name}</td><td>${e.joinDate}</td></tr>`).join('');
   };

   window.selectEmpForSalary = (id, tr) => {
       document.querySelectorAll('#salStatusEmpList tr').forEach(r => r.classList.remove('selected')); tr.classList.add('selected');
       selectedEmpForSalary = emps.find(e => e.empID === id);
       document.getElementById('salCalcFormArea').style.opacity = '1'; document.getElementById('salCalcFormArea').style.pointerEvents = 'auto';
       const keys = ['salBase','salMeal','salDrive','salCar','salService','salRank','salFixedOver'];
       const targetIds = ['scBase','scMeal','scDrive','scCar','scService','scRank','scFixedOver'];
       targetIds.forEach((sid, i) => { document.getElementById(sid).value = parseInt(selectedEmpForSalary[keys[i]] || 0).toLocaleString(); });
       window.updateSalaryStatusCalc(); window.renderSalaryHistory();
   };

   window.updateSalaryStatusCalc = () => {
       if(!selectedEmpForSalary) return;
       const ids = ['scBase','scMeal','scDrive','scCar','scService','scRank','scFixedOver'];
       let total = 0, ordinary = 0;
       ids.forEach(id => { const v = parseInt(document.getElementById(id).value.replace(/,/g,'')) || 0; total += v; if(id !== 'scFixedOver') ordinary += v; });
       const mh = Math.ceil((parseFloat(selectedEmpForSalary.salWorkDay)||8) * (209/8));
       document.getElementById('scTotalMonth').value = total.toLocaleString() + "원";
       document.getElementById('scTotalYear').value = (total * 12).toLocaleString() + "원";
       document.getElementById('scHourPay').value = (mh > 0 ? Math.floor(ordinary/mh) : 0).toLocaleString() + "원";
   };

   window.saveSalaryUpdate = async () => {
       if(!selectedEmpForSalary) return;
       const d = document.getElementById('salDate').value, r = document.getElementById('salReason').value;
       if(!d) return window.showMessageBox("산정일 입력 필수");
       const totalM = document.getElementById('scTotalMonth').value;
       const updateObj = {
           salBase: document.getElementById('scBase').value.replace(/,/g,''), salMeal: document.getElementById('scMeal').value.replace(/,/g,''),
           salDrive: document.getElementById('scDrive').value.replace(/,/g,''), salCar: document.getElementById('scCar').value.replace(/,/g,''),
           salService: document.getElementById('scService').value.replace(/,/g,''), salRank: document.getElementById('scRank').value.replace(/,/g,''),
           salFixedOver: document.getElementById('scFixedOver').value.replace(/,/g,''),
           salaryHistory: arrayUnion({ date: d, reason: r, monthly: totalM, annual: document.getElementById('scTotalYear').value, period: d })
       };
       await updateDoc(doc(empRef, selectedEmpForSalary.empID), updateObj); window.showMessageBox("연봉 정보 업데이트 완료");
       window.renderSalaryHistory();
   };

   window.renderSalaryHistory = () => {
       const b = document.getElementById('salHistoryBody'); const h = selectedEmpForSalary.salaryHistory || [];
       if(h.length === 0) { b.innerHTML = '<tr><td colspan="5">이력 없음</td></tr>'; return; }
       b.innerHTML = h.sort((a,b)=>b.date.localeCompare(a.date)).map(i => `<tr><td>${i.reason}</td><td>${i.date}</td><td>${i.monthly}</td><td>${i.annual}</td><td>-</td></tr>`).join('');
   };

   // ==========================================
   // 6. APPOINTMENT LOGIC
   // ==========================================
   window.renderAppointList = () => {
       const b = document.getElementById('appointEmpList'), s = document.getElementById('appointSearch');
       if(!b || !s) return;
       const f = emps.filter(e => ['정규직','계약직'].includes(e.contractType) && (e.name.includes(s.value) || e.empID.includes(s.value)));
       b.innerHTML = f.map(e => `<tr onclick="window.selectEmpForAppoint('${e.empID}', this)"><td>${e.empID}</td><td>${e.name}</td><td>${e.joinDate}</td></tr>`).join('');
   };

   window.selectEmpForAppoint = (id, tr) => {
       document.querySelectorAll('#appointEmpList tr').forEach(r => r.classList.remove('selected')); tr.classList.add('selected');
       selectedEmpForAppoint = emps.find(e => e.empID === id);
       document.getElementById('appointFormArea').style.opacity = '1'; document.getElementById('appointFormArea').style.pointerEvents = 'auto';
       window.renderAppointHistory();
   };

   window.syncAppointOptions = () => {
       const t = document.getElementById('appType').value, bc = document.getElementById('appBeforeContainer'), ac = document.getElementById('appAfterContainer');
       if(!selectedEmpForAppoint || !bc || !ac) return;
       if(t === '직책부여') {
           bc.innerHTML = `<input id="appBefore" value="${selectedEmpForAppoint.duty || '직책없음'}" readonly style="background:#f9f9f9;">`;
           ac.innerHTML = `<select id="appAfter"><option>대표이사</option><option>본부장</option><option>팀장</option><option>파트장</option><option>직책없음</option></select>`;
       } else if(t === '직위승진') {
           bc.innerHTML = `<input id="appBefore" value="${selectedEmpForAppoint.rank || '사원'}" readonly style="background:#f9f9f9;">`;
           ac.innerHTML = `<select id="appAfter"><option>대표이사</option><option>이사</option><option>수석부장</option><option>부장</option><option>책임</option><option>전임</option><option>선임</option><option>주임</option><option>사원</option></select>`;
       } else if(t.includes('부서')) {
           bc.innerHTML = `<input id="appBefore" value="${selectedEmpForAppoint.dept || ''}" readonly style="background:#f9f9f9;">`; ac.innerHTML = `<input id="appAfter" placeholder="신규 입력">`;
       } else { bc.innerHTML = `<input id="appBefore" readonly style="background:#f9f9f9;">`; ac.innerHTML = `<input id="appAfter" readonly>`; }
   };

   window.saveAppointment = async () => {
       const d = document.getElementById('appDate').value, t = document.getElementById('appType').value, a = document.getElementById('appAfter').value;
       if(!d || !t || !a) return window.showMessageBox("필수 정보 누락");
       const updateObj = { appointments: arrayUnion({ date: d, type: t, before: document.getElementById('appBefore').value, after: a, wage: document.getElementById('appWage').value }) };
       if(t === '직책부여') updateObj.duty = a; else if(t === '직위승진') updateObj.rank = a; else if(t.includes('부서')) updateObj.dept = a;
       await updateDoc(doc(empRef, selectedEmpForAppoint.empID), updateObj); window.showMessageBox("발령 완료");
       window.renderAppointHistory();
   };

   window.renderAppointHistory = () => {
       const b = document.getElementById('appointHistoryBody'); const h = selectedEmpForAppoint.appointments || [];
       if(h.length === 0) { b.innerHTML = '<tr><td colspan="5">내역 없음</td></tr>'; return; }
       b.innerHTML = h.sort((a,b)=>b.date.localeCompare(a.date)).map(i => `<tr><td>${i.date}</td><td>${i.type}</td><td>${i.before}</td><td>${i.after}</td><td>${i.wage}</td></tr>`).join('');
   };

   // ==========================================
   // 8. EVENT LISTENERS & INITIALIZATION
   // ==========================================
   document.addEventListener('input', e => {
       const t = e.target;
       if (t.id === 'salaryPwd') return;

       if (t.dataset.type === 'phone') {
           let v = t.value.replace(/[^0-9]/g, '');
           if (v.startsWith('02')) {
               if(v.length <= 2) t.value = v;
               else if(v.length <= 5) t.value = v.slice(0,2) + '-' + v.slice(2,5) + '-' + v.slice(5);
               else t.value = v.slice(0,2) + '-' + v.slice(2,6) + '-' + v.slice(6,10);
           } else {
               if(v.length <= 3) t.value = v;
               else if(v.length <= 7) t.value = v.slice(0,3) + '-' + v.slice(3,7) + '-' + v.slice(7,11);
               else t.value = v.slice(0,3) + '-' + v.slice(3,7) + '-' + v.slice(7,11);
           }
           return;
       }

       if (t.dataset.type === 'account') {
           window.formatAccount();
           return;
       }

       if (t.classList.contains('salary-input') || (t.id && t.id.startsWith('sal') && !['salWorkDay','salProbationRate'].includes(t.id))) {
           let v = t.value.replace(/[^0-9]/g, ''); t.value = v.replace(/\B(?=(\d{3})+(?!\d))/g, ",");
           window.updateSalaryCalc(); return;
       }
       if (t.id && t.id.startsWith('sc') && !['scHourPay','scTotalMonth','scTotalYear'].includes(t.id)) {
           let v = t.value.replace(/[^0-9]/g, ''); t.value = v.replace(/\B(?=(\d{3})+(?!\d))/g, ",");
           window.updateSalaryStatusCalc(); return;
       }

       if(!t.classList.contains('auto-hyphen')) return;
       let v = t.value.replace(/[^0-9]/g, '');
       if(t.dataset.type === 'ssn') t.value = v.length <= 6 ? v : v.slice(0,6)+'-'+v.slice(6,13);
       else if(t.dataset.type === 'date') t.value = v.length <= 4 ? v : v.length <= 6 ? v.slice(0,4)+'-'+v.slice(4) : v.slice(0,4)+'-'+v.slice(4,6)+'-'+v.slice(6,8);
       else if(t.dataset.type === 'year-month') t.value = v.length <= 4 ? v : v.slice(0,4)+'-'+v.slice(4,6);
       else if(t.dataset.type === 'date-range') t.value = v.length <= 4 ? v : v.length <= 6 ? v.slice(0,4)+'.'+v.slice(4) : v.length <= 10 ? v.slice(0,4)+'.'+v.slice(4,6)+' ~ '+v.slice(6,10) : v.slice(0,4)+'.'+v.slice(4,6)+' ~ '+v.slice(6,10)+'.'+v.slice(10,12);
   });

   async function start() {
       await signInAnonymously(auth);
       onAuthStateChanged(auth, user => {
           if(user) onSnapshot(empRef, snap => {
               emps = snap.docs.map(d => ({...d.data(), id: d.id}));
               window.renderGrid();
               window.renderSalaryStatusList();
               window.renderAppointList();
           });
       });
   }

   start();
